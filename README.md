    <script>
        const fixedBackHome = document.getElementById('fixed-back-home');
        const fixedBackToTop = document.getElementById('fixed-back-to-top');
        
        fixedBackHome.addEventListener('click', () => {
            window.location.href = 'index.php';
        });
        
        fixedBackToTop.addEventListener('click', () => {
            window.scrollTo({
                top: 0,
                behavior: 'smooth'
            });
        });
        
        const bookmarkBtn = document.getElementById('bookmark-btn');
        
        bookmarkBtn.addEventListener('click', () => {
            const url = window.location.href;
            const title = document.title;
            
            // 检测是否为iOS设备
            const isIOS = /iPhone|iPad|iPod/i.test(navigator.userAgent);
            // 检测是否为Android设备
            const isAndroid = /Android/i.test(navigator.userAgent);
            
            if (isIOS) {
                // iOS设备提示
                alert('请点击底部分享按钮 📤，然后选择"添加到主屏幕"或"添加书签"');
            } else if (isAndroid) {
                // Android设备提示
                alert('请点击浏览器菜单按钮（三个点 ⋮），然后选择"收藏"或"添加书签"');
            } else {
                // 桌面浏览器
                try {
                    // 尝试使用现代浏览器API
                    if (typeof window.external !== 'undefined' && 'AddFavorite' in window.external) {
                        // IE浏览器
                        window.external.AddFavorite(url, title);
                    } else if (window.sidebar && typeof window.sidebar.addPanel === 'function') {
                        // Firefox旧版本
                        window.sidebar.addPanel(title, url, '');
                    } else {
                        // 其他浏览器显示快捷键提示
                        const isMac = /Mac/i.test(navigator.userAgent);
                        const shortcut = isMac ? 'Command+D' : 'Ctrl+D';
                        alert('请按 ' + shortcut + ' 键将本页添加到收藏夹');
                    }
                } catch (e) {
                    // 如果出错，显示通用提示
                    alert('请使用浏览器的收藏功能添加本页到书签');
                }
            }
        });
        
        const themeToggle = document.getElementById('theme-toggle');
        const body = document.body;
        
        // 初始化主题状态
        const savedTheme = localStorage.getItem('theme');
        if (savedTheme === 'dark') {
            body.classList.add('dark-theme');
            themeToggle.querySelector('.btn-icon').innerHTML = '🌙';
            themeToggle.querySelector('.btn-text').textContent = '夜间';
        } else {
            body.classList.remove('dark-theme');
            themeToggle.querySelector('.btn-icon').innerHTML = '🌞';
            themeToggle.querySelector('.btn-text').textContent = '日间';
        }
        
        // 主题切换事件
        themeToggle.addEventListener('click', () => {
            body.classList.toggle('dark-theme');
            if (body.classList.contains('dark-theme')) {
                themeToggle.querySelector('.btn-icon').innerHTML = '🌙';
                themeToggle.querySelector('.btn-text').textContent = '夜间';
                localStorage.setItem('theme', 'dark');
            } else {
                themeToggle.querySelector('.btn-icon').innerHTML = '🌞';
                themeToggle.querySelector('.btn-text').textContent = '日间';
                localStorage.setItem('theme', 'light');
            }
        });
        
        const backToTop = document.getElementById('fixed-back-to-top');
        
        backToTop.classList.add('show');
        
        backToTop.addEventListener('click', () => {
            window.scrollTo({
                top: 0,
                behavior: 'smooth'
            });
        });
        
        const stepItems = document.querySelectorAll('.step-item');

        stepItems.forEach(item => {
            item.addEventListener('click', () => {
                item.classList.toggle('active');
            });
        });
    </script>
